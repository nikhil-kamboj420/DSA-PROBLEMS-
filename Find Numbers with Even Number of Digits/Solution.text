class Solution {
    public int findNumbers(int[] nums) {

        int countEven = 0;

        for (int i = 0; i < nums.length; i++) {

            int digits = nums[i];
            int count = 0;

            while (digits != 0) {
                count++;
                digits /= 10;
            }

            if (count % 2 == 0) {
                countEven++;
            }
        }

        return countEven;
    }
}
