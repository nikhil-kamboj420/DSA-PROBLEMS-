class Solution {
    public int maxProfit(int[] prices) {
        int left = 0;
        int maxProfit = 0;

        for (int right = 1; right < prices.length; right++) {
            if (prices[right] > prices[left]) {
                int curProfit = prices[right] - prices[left];
                maxProfit = Math.max(maxProfit, curProfit);
            } else {
                left = right;
            }
        }

        return maxProfit;

    }
}
